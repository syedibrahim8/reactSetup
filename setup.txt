1.git clone <your project repo>
2.cd <repo>
3.mkdir client && cd client
4.pnpm create vite@latest .

5.choose: framework -> react ; variant:Javascript ;

6. rm -rf eslint.config.js && src/assets/react.svg && src/App.css && src/App.jsx && src/index.css && src/main.jsx

7. touch src/App.css && src/App.jsx && src/index.css && src/main.jsx

8.pnpm install && pnpm up --latest

9.pnpm add tailwindcss @tailwindcss/vite

10. import tailwindcss in src/index.css

11. In vite.config.jsx
    import {defineConfig} from 'vite'
    import react from '@vitejs/plugin-react'
    import tailwindcss from '@tailwindcss/vite'

    export default defineConfig({
        plugins:[react(),tailwindcss()]
    })

12. In src/main.jsx
    import {createRoot} from 'react-dom/client';
    import './index.css';
    import App from './App.jsx';

    createRoot(document.getElementById('root')).render(<App />);

13. In src/App.jsx
    import React from 'react';
    const App = () =>{
        return(
            <h1 "className = text-3xl font-bold text-center text-gray-800 mt-10">
            setup is completed
            <h1 />
        )
    }

    export default App

14. pnpm dev
